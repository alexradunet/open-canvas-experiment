ES Modules — Quick Reference

Entry point

<script type="module" src="main.js"></script>

Exports

// Named exports (multiple per module)
export const VERSION = '1.0';
export function normalize(path) { /* ... */ }
export class VaultStore { /* ... */ }

// Export list (apart from declarations)
const a = 1, b = 2;
export { a, b };

// Rename on export
export { normalize as normalizePath };

// Default export (one per module)
export default class App { /* ... */ }

Imports

// Named imports
import { normalize, VERSION } from './storage/vault-path.js';

// Rename on import
import { normalize as norm } from './storage/vault-path.js';

// Namespace import
import * as vault from './storage/vault-store.js';
vault.normalize('/foo/');

// Default import
import App from './app.js';

// Side-effect only (no bindings)
import './styles/init.js';

// Dynamic (runtime, returns Promise)
const mod = await import('./heavy-feature.js');
mod.doThing();

Re-exports

// Pass-through (not available in current file)
export { normalize } from './storage/vault-path.js';

// Rename
export { normalize as norm } from './storage/vault-path.js';

// Re-export default
export { default } from './storage/vault-store.js';
export { default as VaultStore } from './storage/vault-store.js';

Key behaviors

BehaviorDetail
ScopeEach module has its own top-level scope. No globals leak.
Strict modeAlways strict. No opt-out.
ExecutionOnce per module. Cached in module registry.
BindingsLive — importers see updated exported values.
PathsMust be full URLs with extension. No bare specifiers.
DeferredModule scripts execute after HTML parsing (like defer).
CORSCross-origin modules need CORS headers.

import.meta

// URL of the current module file
const moduleURL = import.meta.url;
// e.g. "http://localhost:4173/storage/vault-path.js"

// Resolve paths relative to this module
const workerURL = new URL('./worker.js', import.meta.url);

Common patterns (no-build)

// Barrel file (re-export public API)
// storage/index.js
export { MemoryVault } from './memory-vault.js';
export { FsVault } from './fs-vault.js';
export { IndexedDbVault } from './indexeddb-vault.js';

// Consumer imports from barrel
import { MemoryVault } from './storage/index.js';

// Conditional loading
if (needsHeavyFeature) {
  const { HeavyWidget } = await import('./widgets/heavy.js');
  mount(new HeavyWidget());
}
